Usage & Code Examples

# Install if not already
pip install npmai

# Basic usage
from npmai import Ollama

llm = Ollama(
        model="llama3.2",
        temperature=0.7
)
prompts = "Hello, tell me a short summary of NPMAI"

result = llm.invoke(prompts)
print(result)
      
# Using temperature
from npmai import Ollama

llm = Ollama(
        model="llama3.2",
        temperature=0.7 # here passs your temperature parameter
)
prompts = "Hello, tell me a short summary of NPMAI"

result = llm.invoke(prompts)
print(result)
      
# Validated Schemas
from npmai import Ollama

llm = Ollama(
        model="llama3.2",
        temperature=0.7,
        validate_schema=#Here pass yor schema
)
prompts = "Hello, tell me a short summary of NPMAI"

result = llm.invoke(prompts)
print(result)
